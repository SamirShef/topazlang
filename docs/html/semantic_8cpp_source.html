21void SemanticAnalyzer::analyze_stmt(AST::Stmt& stmt) {
22 if (auto vds = dynamic_cast<AST::VarDeclStmt*>(&stmt)) {
23 analyze_var_decl_stmt(*vds);
24 }
25 else if (auto vas = dynamic_cast<AST::VarAsgnStmt*>(&stmt)) {
26 analyze_var_asgn_stmt(*vas);
27 }
28 else if (auto fds = dynamic_cast<AST::FuncDeclStmt*>(&stmt)) {
29 analyze_func_decl_stmt(*fds);
30 }
31 else if (auto fcs = dynamic_cast<AST::FuncCallStmt*>(&stmt)) {
32 analyze_func_call_stmt(*fcs);
33 }
34 else if (auto rs = dynamic_cast<AST::ReturnStmt*>(&stmt)) {
35 analyze_return_stmt(*rs);
36 }
37 else if (auto ies = dynamic_cast<AST::IfElseStmt*>(&stmt)) {
38 analyze_if_else_stmt(*ies);
39 }
40 else {
41 throw_exception(SUB_SEMANTIC, "Unsupported statement. Please check your Topaz compiler version and fix the problematic section of the code", stmt.line, file_name);
42 }

45void SemanticAnalyzer::analyze_var_decl_stmt(AST::VarDeclStmt& vds) {
46 std::unique_ptr<Value> value = get_variable_value(vds.name);
47 if (value != nullptr) {
48 std::stringstream ss;
49 ss << "Variable \033[0m'" << vds.name << "'\033[31m already exists";
50 throw_exception(SUB_SEMANTIC, ss.str(), vds.line, file_name);
51 }
52 AST::Type var_type = vds.type;
53 Value var_val = Value(var_type, get_default_val_by_type(var_type, vds.line));

56 }
57 if (!has_common_type(var_val.type, var_type)) {
58 std::stringstream ss;
59 ss << "Type mismatch: an expression of the type \033[0m'" << var_val.type.to_str() << "'\033[31m, but the type is expected \033[0m'" << var_type.to_str() << "'\033[31m";
60 throw_exception(SUB_SEMANTIC, ss.str(), vds.line, file_name);
61 }
62 variables.top().emplace(vds.name, var_val);

191 right_type.type >= AST::TYPE_BOOL && right_type.type <= AST::TYPE_DOUBLE && left_type.type > AST::TYPE_DOUBLE) {
192 std::stringstream ss;
193 ss << "Type mismatch: it is not possible to use the binary \033[0m'" << be.op.value <<"'\033[31m operator with \033[0m'" << left_type.to_str() << "'\033[31m and \033[0m'" << right_type.to_str() <<"'\033[31m types";

198 if (be.op.type != TOK_OP_PLUS) {
199 std::stringstream ss;
200 ss << "Type mismatch: it is not possible to use the binary \033[0m'" << be.op.value <<"'\033[31m operator with \033[0m'" << left_type.to_str() << "'\033[31m and \033[0m'" << right_type.to_str() <<"'\033[31m types";

218 }
219 case TOK_OP_EQ_EQ:

222 case TOK_OP_GT_EQ:
223 case TOK_OP_LS:
224 case TOK_OP_LS_EQ:

336SemanticAnalyzer::Value SemanticAnalyzer::get_function_return_value(FunctionInfo *func, AST::FuncCallExpr& fce) {
337 variables.push({});
338 for (size_t i = 0; i < fce.args.size(); i++) {
339 variables.top().emplace(func->args[i].name, analyze_expr(*fce.args[i]));
340 }
341 for (auto& stmt : func->block) {
342 if (auto rs = dynamic_cast<AST::ReturnStmt*>(&*stmt)) {
343 Value val = analyze_expr(*rs->expr);
344 variables.pop();
345 return val;
346 }
347 else if (auto ies = dynamic_cast<AST::IfElseStmt*>(&*stmt)) {
348 Value *val = get_function_return_value_from_if_else(*ies);
349 if (val != nullptr) {
350 variables.pop();
351 return *val;
352 }
353 }
354 }
355 std::stringstream ss;
356 ss << "Function \033[0m'" << fce.name << "'\033[31m does not returning value. Please add \033[0m'return'\033[31m statement into the end of the function";
357 throw_exception(SUB_SEMANTIC, ss.str(), fce.line, file_name);
358}

360SemanticAnalyzer::Value *SemanticAnalyzer::get_function_return_value_from_if_else(AST::IfElseStmt& ies) {
361 Value cond_val = analyze_expr(*ies.cond);
362 if (std::get<bool>(cond_val.value.value) == true) {
363 for (auto& stmt : ies.then_block) {
364 if (auto rs = dynamic_cast<AST::ReturnStmt*>(&*stmt)) {
365 static Value val = analyze_expr(*rs->expr);
366 return &val;
367 }
368 else if (auto ies = dynamic_cast<AST::IfElseStmt*>(&*stmt)) {
369 return get_function_return_value_from_if_else(*ies);
370 }
371 }
372 }
373 else {
374 for (auto& stmt : ies.else_block) {
375 if (auto rs = dynamic_cast<AST::ReturnStmt*>(&*stmt)) {
376 static Value val = analyze_expr(*rs->expr);
377 return &val;
378 }
379 else if (auto ies = dynamic_cast<AST::IfElseStmt*>(&*stmt)) {
380 return get_function_return_value_from_if_else(*ies);
381 }
382 }
383 }

430bool SemanticAnalyzer::has_common_type(AST::Type left, AST::Type right) {
431 if (left.type == right.type) {
432 return true;
433 }
434 if (implicitly_cast_allowed_types.find(left.type) != implicitly_cast_allowed_types.end() &&
435 std::find(implicitly_cast_allowed_types[left.type].begin(), implicitly_cast_allowed_types[left.type].end(), right.type) != implicitly_cast_allowed_types[left.type].end()) {
436 return true;
437 }
438 return false;
439}

441AST::Type SemanticAnalyzer::get_common_type(AST::Type left, AST::Type right, uint32_t line) {
442 if (left.type == right.type) {
443 return left;
444 }
445
446 if (has_common_type(left, right)) {
447 return AST::Type(implicitly_cast_allowed_types[left.type][right.type], right.name, right.is_const, right.is_ptr, right.is_nullable);
448 }
449
450 std::stringstream ss;
451 ss << "Type mismatch: there is no common type for \033[0m'" << left.to_str() << "'\033[31m and \033[0m'" << right.to_str() << "'\033[31m";
452 throw_exception(SUB_SEMANTIC, ss.str(), line, file_name);
453}

455double SemanticAnalyzer::binary_two_variants(Value left, Value right, TokenType op, uint32_t line) {
456 double left_val = 0;
457 double right_val = 0;
458 switch (left.type.type) {
459 case AST::TYPE_BOOL:
460 left_val = std::get<0>(left.value.value);
461 break;
462 case AST::TYPE_CHAR:
463 left_val = std::get<1>(left.value.value);
464 break;
465 case AST::TYPE_SHORT:
466 left_val = std::get<2>(left.value.value);
467 break;
468 case AST::TYPE_INT:
469 left_val = std::get<3>(left.value.value);
470 break;
471 case AST::TYPE_LONG:
472 left_val = std::get<4>(left.value.value);
473 break;
474 case AST::TYPE_FLOAT:
475 left_val = std::get<5>(left.value.value);
476 break;
477 case AST::TYPE_DOUBLE:
478 left_val = std::get<6>(left.value.value);
479 break;
480 }
481 switch (right.type.type) {
482 case AST::TYPE_BOOL:
483 right_val = std::get<0>(right.value.value);
484 break;
485 case AST::TYPE_CHAR:
486 right_val = std::get<1>(right.value.value);
487 break;
488 case AST::TYPE_SHORT:
489 right_val = std::get<2>(right.value.value);
490 break;
491 case AST::TYPE_INT:
492 right_val = std::get<3>(right.value.value);
493 break;
494 case AST::TYPE_LONG:
495 right_val = std::get<4>(right.value.value);
496 break;
497 case AST::TYPE_FLOAT:
498 right_val = std::get<5>(right.value.value);
499 break;
500 case AST::TYPE_DOUBLE:
501 right_val = std::get<6>(right.value.value);
502 break;
503 }
504 switch (op) {
505 case TOK_OP_PLUS:
506 return left_val + right_val;
507 case TOK_OP_MINUS:
508 return left_val - right_val;
509 case TOK_OP_MULT:
510 return left_val * right_val;
511 case TOK_OP_DIV:
512 if (right_val == 0) {
513 throw_exception(SUB_SEMANTIC, "Division by zero", line, file_name);
514 }
515 return left_val / right_val;
516 case TOK_OP_MODULO:
517 return std::fmod(left_val, right_val);
518 case TOK_OP_EQ_EQ:
519 return static_cast<bool>(left_val == right_val);
520 case TOK_OP_NOT_EQ_EQ:
521 return static_cast<bool>(left_val != right_val);
522 case TOK_OP_GT:
523 return static_cast<bool>(left_val > right_val);
524 case TOK_OP_GT_EQ:
525 return static_cast<bool>(left_val >= right_val);
526 case TOK_OP_LS:
527 return static_cast<bool>(left_val < right_val);
528 case TOK_OP_LS_EQ:
529 return static_cast<bool>(left_val <= right_val);
