12std::vector<Token> Lexer::tokenize() {
13 std::vector<Token> tokens;
14
15 while (pos < source_len) {
16 const char c = peek();
17 if (c == ' ' || c == '\n') {
18 advance();
19 }
20 else if (c == '/') {
21 if (peek(1) == '/') {
22 skip_comments();
23 }
24 else {
25 tokens.push_back(tokenize_op());
26 }
27 }
28 else if (isalpha(c)) {
29 tokens.push_back(tokenize_id());
30 }
31 else if (isdigit(c)) {
32 tokens.push_back(tokenize_number_lit());
33 }
34 else if (c == '\"') {
35 tokens.push_back(tokenize_string_lit());
36 }
37 else if (c == '\'') {
38 tokens.push_back(tokenize_character_lit());
39 }
40 else {
41 tokens.push_back(tokenize_op());
42 }
43 }
44
45 return tokens;
46}

48Token Lexer::tokenize_id() {
49 std::string value;
50 uint32_t tmp_l = line;
51 uint32_t tmp_c = column;
52
53 while (pos < source_len && (isalpha(peek()) || isdigit(peek()) || peek() == '_')) {
54 value += advance();
55 }
56
57 if (keywords.find(value) != keywords.end()) {
58 return Token(keywords[value], value, tmp_l, tmp_c, file_name);
59 }
60 else if (value == "true" || value == "false") {
61 return Token(TOK_BOOLEAN_LIT, value, tmp_l, tmp_c, file_name);
62 }
63 return Token(TOK_ID, value, tmp_l, tmp_c, file_name);

66Token Lexer::tokenize_number_lit() {
67 std::string value;
68 uint32_t tmp_l = line;
69 uint32_t tmp_c = column;
70 bool has_dot = false;
71
72 while (pos < source_len && (isdigit(peek()) || peek() == '.' || peek() == '_')) {
73 if (peek() == '_') {
74 advance();
75 continue;
76 }
77 else if (peek() == '.') {
78 if (has_dot) {
79 throw_excpetion(SUB_LEXER, "Invalid number literal: twice dot", line, file_name);
80 }
81 else if (pos < source_len && peek(1) == '_') {
82 throw_excpetion(SUB_LEXER, "Invalid number literal: \033[0m'_'\033[31m cannot be immediately after the dot", line, file_name);
83 }
84 else if (pos < source_len && !isdigit(peek(1))) {
85 throw_excpetion(SUB_LEXER, "Invalid number literal: dot cannot be the end", line, file_name);
86 }
87 has_dot = true;
88 }
89 value += advance();
90 }
91
92 const char suffix = pos < source_len ? peek() : '\0';
93 switch (tolower(suffix)) {
94 case 'f':
95 advance();
96 return Token(TOK_FLOAT_LIT, value, tmp_l, tmp_c, file_name);
97 case 's':
98 if (has_dot) {
99 throw_excpetion(SUB_LEXER, "Invalid number literal: specified suffix \033[0m's'\033[31m does not match for floating point literal", line, file_name);
100 }
101 advance();
102 return Token(TOK_SHORT_LIT, value, tmp_l, tmp_c, file_name);
103 case 'l':
104 if (has_dot) {
105 throw_excpetion(SUB_LEXER, "Invalid number literal: specified suffix \033[0m'l'\033[31m does not match for floating point literal", line, file_name);
106 }
107 advance();
108 return Token(TOK_LONG_LIT, value, tmp_l, tmp_c, file_name);
109 default:
110 if (has_dot) {
111 return Token(TOK_DOUBLE_LIT, value, tmp_l, tmp_c, file_name);
112 }
113 else {
114 return Token(TOK_INT_LIT, value, tmp_l, tmp_c, file_name);
115 }
116 }
117}

164Token Lexer::tokenize_op() {
165 uint32_t tmp_l = line;
166 uint32_t tmp_c = column;
167 const char c = advance();
168
169 switch (c) {
170 case '+':
171 if (peek() == '=') {
172 advance();
173 return Token(TOK_OP_PLUS_EQ, "+=", tmp_l, tmp_c, file_name);
174 }
175 return Token(TOK_OP_PLUS, "+", tmp_l, tmp_c, file_name);
176 case '-':
177 if (peek() == '=') {
178 advance();
179 return Token(TOK_OP_MINUS_EQ, "-=", tmp_l, tmp_c, file_name);
180 }
181 else if (peek() == '>') {
182 advance();
183 return Token(TOK_OP_NEXT, "->", tmp_l, tmp_c, file_name);
184 }
185 return Token(TOK_OP_MINUS, "-", tmp_l, tmp_c, file_name);
186 case '*':
187 if (peek() == '=') {
188 advance();
189 return Token(TOK_OP_MULT_EQ, "*=", tmp_l, tmp_c, file_name);
190 }
191 return Token(TOK_OP_MULT, "*", tmp_l, tmp_c, file_name);
192 case '/':
193 if (peek() == '=') {
194 advance();
195 return Token(TOK_OP_DIV_EQ, "/=", tmp_l, tmp_c, file_name);
196 }
197 return Token(TOK_OP_DIV, "/", tmp_l, tmp_c, file_name);
198 case '%':
199 if (peek() == '=') {
200 advance();
201 return Token(TOK_OP_MODULO_EQ, "%=", tmp_l, tmp_c, file_name);
202 }
203 return Token(TOK_OP_MODULO, "%", tmp_l, tmp_c, file_name);
204 case '=':
205 if (peek() == '=') {
206 advance();
207 return Token(TOK_OP_EQ_EQ, "==", tmp_l, tmp_c, file_name);
208 }
209 return Token(TOK_OP_EQ, "=", tmp_l, tmp_c, file_name);
210 case '!':
211 if (peek() == '=') {
212 advance();
213 return Token(TOK_OP_NOT_EQ_EQ, "!=", tmp_l, tmp_c, file_name);
214 }
215 return Token(TOK_OP_L_NOT, "!", tmp_l, tmp_c, file_name);
216 case '>':
