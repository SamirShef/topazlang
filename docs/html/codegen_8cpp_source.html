33void CodeGenerator::generate_stmt(AST::Stmt& stmt) {
34 if (auto vds = dynamic_cast<AST::VarDeclStmt*>(&stmt)) {
35 generate_var_decl_stmt(*vds);
36 }
37 else if (auto vas = dynamic_cast<AST::VarAsgnStmt*>(&stmt)) {
38 generate_var_asgn_stmt(*vas);
39 }
40 else if (auto fds = dynamic_cast<AST::FuncDeclStmt*>(&stmt)) {
41 generate_func_decl_stmt(*fds);
42 }
43 else if (auto rs = dynamic_cast<AST::ReturnStmt*>(&stmt)) {
44 generate_return_stmt(*rs);
45 }
46 else {
47 throw_exception(SUB_CODEGEN, "Unsupported statement. Please check your Topaz compiler version and fix the problematic section of the code", stmt.line, file_name);
48 }

51void CodeGenerator::generate_var_decl_stmt(AST::VarDeclStmt& vds) {
52 llvm::Type *type = type_to_llvm(vds.type);
53 llvm::Value *val = llvm::Constant::getNullValue(type);
54 if (vds.expr != nullptr) {
55 val = generate_expr(*vds.expr);
56 }
57 llvm::Value *var = nullptr;
58 if (variables.size() == 1) {
59 var = new llvm::GlobalVariable(*module, type, vds.type.is_const, llvm::GlobalValue::ExternalLinkage, llvm::dyn_cast<llvm::Constant>(val), vds.name);
60 }
61 else {
62 var = builder.CreateAlloca(type, nullptr, vds.name + ".alloca");
63 builder.CreateStore(val, var);
64 }
65 variables.top().emplace(vds.name, var);
66}

68void CodeGenerator::generate_var_asgn_stmt(AST::VarAsgnStmt& vas) {
69 llvm::Value *var_inst = nullptr;
70 auto vars = variables;
71 while (!vars.empty()) {
72 auto vars_it = vars.top().find(vas.name);
73 if (vars_it != vars.top().end()) {
74 var_inst = vars_it->second;
75 }
76 vars.pop();
77 }
78 if (var_inst == nullptr) {
79 std::stringstream ss;
80 ss << "Variable \033[0m'" << vas.name << "'\033[31m does not exists";
81 throw_exception(SUB_CODEGEN, ss.str(), vas.line, file_name);
82 }
83 builder.CreateStore(generate_expr(*vas.expr), var_inst);
84}

86void CodeGenerator::generate_func_decl_stmt(AST::FuncDeclStmt& fds) {
87 llvm::Type *ret_type = type_to_llvm(fds.ret_type);
88 std::vector<llvm::Type*> args;
89 size_t args_count = fds.args.size();
90 for (size_t i = 0; i < args_count; i++) {
91 args.push_back(type_to_llvm(fds.args[i].type));
92 }
93 llvm::FunctionType *func_type = llvm::FunctionType::get(ret_type, args, false);
94 llvm::Function *func = llvm::Function::Create(func_type, llvm::GlobalValue::ExternalLinkage, fds.name, *module);
95
96 llvm::BasicBlock *entry = llvm::BasicBlock::Create(context, "entry", func);
97 builder.SetInsertPoint(entry);
98
99 variables.push({});
100 functions.emplace(fds.name, func);
101 size_t index = 0;
102 for (llvm::Argument& arg : func->args()) {
103 arg.setName(fds.args[index].name);
104 llvm::AllocaInst* arg_alloca = builder.CreateAlloca(arg.getType(), nullptr, fds.args[index].name);
105 builder.CreateStore(&arg, arg_alloca);
106 variables.top().emplace(fds.args[index].name, arg_alloca);
107 index++;
108 }
109 size_t block_size = fds.block.size();
110 for (size_t i = 0; i < block_size; i++) {
111 generate_stmt(*fds.block[i]);
112 }
113 variables.pop();
114}

143llvm::Value *CodeGenerator::generate_literal_expr(AST::Literal& lit) {
144 auto& value = lit.value.value;
145
146 switch (lit.type.type) {
147 case AST::TYPE_CHAR:
148 return llvm::ConstantInt::get(type_to_llvm(lit.type), llvm::APInt(8, std::get<char8_t>(value)));
149 case AST::TYPE_SHORT:
150 return llvm::ConstantInt::get(type_to_llvm(lit.type), llvm::APInt(16, std::get<int16_t>(value)));
151 case AST::TYPE_INT:
152 return llvm::ConstantInt::get(type_to_llvm(lit.type), llvm::APInt(32, std::get<int32_t>(value)));
153 case AST::TYPE_LONG:
154 return llvm::ConstantInt::get(type_to_llvm(lit.type), llvm::APInt(64, std::get<int64_t>(value)));
155 case AST::TYPE_FLOAT:
156 return llvm::ConstantFP::get(type_to_llvm(lit.type), llvm::APFloat(std::get<float_t>(value)));
157 case AST::TYPE_DOUBLE:
158 return llvm::ConstantFP::get(type_to_llvm(lit.type), llvm::APFloat(std::get<double_t>(value)));
159 case AST::TYPE_BOOL:
160 return llvm::ConstantInt::get(type_to_llvm(lit.type), llvm::APInt(1, std::get<bool>(value)));
161 case AST::TYPE_STRING_LIT: {
162 llvm::Constant *str_const = llvm::ConstantDataArray::getString(context, std::get<std::string>(value), true);
163 llvm::GlobalVariable *str_var = new llvm::GlobalVariable(*module, str_const->getType(), true, llvm::GlobalValue::PrivateLinkage, str_const, "string.lit");
164 return str_var;
165 }
166 default:
167 throw_exception(SUB_CODEGEN, "An unsupported literal type was encountered during compilation. Please check your Topaz compiler version and fix the problematic section of the code", lit.line, file_name);
168 }

171llvm::Value *CodeGenerator::generate_binary_expr(AST::BinaryExpr& be) {
172 llvm::Value *left = generate_expr(*be.left_expr);
173 llvm::Type *left_type = left->getType();
174 llvm::Value *right = generate_expr(*be.right_expr);
175 llvm::Type *right_type = right->getType();
176
177 switch (be.op.type) {
178 case TOK_OP_PLUS:
179 if (left_type->isFloatingPointTy() || right_type->isFloatingPointTy()) {
180 return builder.CreateFAdd(left, right, "fadd.tmp");
181 }
182 return builder.CreateAdd(left, right, "add.tmp");
183 case TOK_OP_MINUS:
184 if (left_type->isFloatingPointTy() || right_type->isFloatingPointTy()) {
185 return builder.CreateFSub(left, right, "fsub.tmp");
186 }
187 return builder.CreateSub(left, right, "sub.tmp");
188 case TOK_OP_MULT:
189 if (left_type->isFloatingPointTy() || right_type->isFloatingPointTy()) {
190 return builder.CreateFMul(left, right, "fmul.tmp");
191 }
192 return builder.CreateMul(left, right, "mul.tmp");
193 case TOK_OP_DIV:
194 if (left_type->isFloatingPointTy() || right_type->isFloatingPointTy()) {
195 return builder.CreateFDiv(left, right, "fdiv.tmp");
196 }
197 return builder.CreateSDiv(left, right, "div.tmp");
198 case TOK_OP_MODULO:
199 if (left_type->isFloatingPointTy() || right_type->isFloatingPointTy()) {
200 return builder.CreateFRem(left, right, "frem.tmp");
201 }
202 return builder.CreateSRem(left, right, "rem.tmp");
203 case TOK_OP_EQ_EQ:
204 if (left_type->isFloatingPointTy() || right_type->isFloatingPointTy()) {
205 return builder.CreateFCmpUEQ(left, right, "feq.tmp");
206 }
207 return builder.CreateICmpEQ(left, right, "eq.tmp");
208 case TOK_OP_NOT_EQ_EQ:
209 if (left_type->isFloatingPointTy() || right_type->isFloatingPointTy()) {
210 return builder.CreateNeg(builder.CreateFCmpUEQ(left, right, "feq.tmp"), "fnoteq.tmp");
211 }
212 return builder.CreateNeg(builder.CreateICmpEQ(left, right, "eq.tmp"), "noteq.tmp");
213 case TOK_OP_GT:
214 if (left_type->isFloatingPointTy() || right_type->isFloatingPointTy()) {
215 return builder.CreateFCmpUGT(left, right, "fgt.tmp");
216 }
217 return builder.CreateICmpSGT(left, right, "gt.tmp");
218 case TOK_OP_GT_EQ:
219 if (left_type->isFloatingPointTy() || right_type->isFloatingPointTy()) {
220 return builder.CreateFCmpUGE(left, right, "fge.tmp");
221 }
222 return builder.CreateICmpSGE(left, right, "ge.tmp");
223 case TOK_OP_LS:
224 if (left_type->isFloatingPointTy() || right_type->isFloatingPointTy()) {
225 return builder.CreateFCmpULT(left, right, "flt.tmp");
226 }
227 return builder.CreateICmpSLT(left, right, "lt.tmp");
228 case TOK_OP_LS_EQ:
229 if (left_type->isFloatingPointTy() || right_type->isFloatingPointTy()) {
230 return builder.CreateFCmpULE(left, right, "fle.tmp");
231 }
232 return builder.CreateICmpSLE(left, right, "le.tmp");
233 case TOK_OP_L_AND:
234 return builder.CreateLogicalAnd(left, right, "land.tmp");
235 case TOK_OP_L_OR:
236 return builder.CreateLogicalAnd(left, right, "lor.tmp");
237 default:
238 throw_exception(SUB_CODEGEN, "An unsupported binary operator was encountered during compilation. Please check your Topaz compiler version and fix the problematic section of the code", be.line, file_name);
239 }
