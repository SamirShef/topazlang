168AST::ExprPtr Parser::parse_primary_expr() {
169 Token token = peek();
170 switch (token.type) {
171 case TOK_OP_LPAREN: {
172 AST::ExprPtr expr = parse_expr();
173 consume(TOK_OP_RPAREN, "Expected ')'. You forgot to specify the closing ')'", token.line);
174 return expr;
175 }
176 case TOK_CHARACTER_LIT:
177 pos++;
178 return std::make_unique<AST::CharacterLiteral>(token.value[0], token.line);
179 case TOK_SHORT_LIT:
180 pos++;
181 return std::make_unique<AST::ShortLiteral>(std::stoll(token.value), token.line);
182 case TOK_INT_LIT:
183 pos++;
184 return std::make_unique<AST::IntLiteral>(std::stoll(token.value), token.line);
185 case TOK_LONG_LIT:
186 pos++;
187 return std::make_unique<AST::LongLiteral>(std::stoll(token.value), token.line);
188 case TOK_FLOAT_LIT:
189 pos++;
190 return std::make_unique<AST::FloatLiteral>(std::stold(token.value), token.line);
191 case TOK_DOUBLE_LIT:
192 pos++;
193 return std::make_unique<AST::DoubleLiteral>(std::stold(token.value), token.line);
194 case TOK_BOOLEAN_LIT:
195 pos++;
196 return std::make_unique<AST::BoolLiteral>(token.value == "true", token.line);
197 case TOK_STRING_LIT:
198 pos++;
199 return std::make_unique<AST::StringLiteral>(token.value, token.line);
200 default:
201 throw_excpetion(SUB_PARSER, "Unsupported expression. Please check expression to mistakes", token.line, token.file_name);
202 }
