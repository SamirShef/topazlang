42AST::StmtPtr Parser::parse_var_decl_stmt() {
43 Token first_token = peek(-1);
44 AST::Type type = consume_type();
45 std::stringstream ss;
46 ss << "Expected \033[0m':'\033[31m between type and variable name.\nPlease replace \033[0m'";
47 ss << "let " << type.to_str() << "'\033[31m with: \033[0m'let " << type.to_str() << ": '";
48 consume(TOK_OP_COLON, ss.str(), peek().line);
49
50 ss.str("");
51 ss << "Expected variable name.\nToken \033[0m'" << peek().value << "'\033[31m is keyword or operator. Please replase it with unique identifier";
52 std::string name = consume(TOK_ID, ss.str(), peek().line).value;
53 AST::ExprPtr expr = nullptr;
54 if (pos == tokens_count) {
55 ss.str("");
56 ss << "Expected \033[0m';'\033[31m in the end of variable definition. Please add \033[0m';'\033[31m into the end of variable definition";
57 throw_exception(SUB_PARSER, ss.str(), peek(-1).line, peek(-1).file_name);
58 }
59 if (match(TOK_OP_EQ)) {
60 expr = parse_expr();
61 }
62
63 ss.str("");
64 ss << "Expected \033[0m';'\033[31m in the end of variable definition. ";
65 if (pos == tokens_count) {
66 ss << "Please add \033[0m';'\033[31m into the end of variable definition";
67 }
68 else {
69 ss << "Please replace \033[0m'" << peek().value << "'\033[31m with \033[0m';'";
70 }
71 consume(TOK_OP_SEMICOLON, ss.str(), peek().line);
72
73 return std::make_unique<AST::VarDeclStmt>(type, std::move(expr), name, first_token.line);
74}

76AST::StmtPtr Parser::parse_var_asgn_stmt() {

95 }
96 consume(TOK_OP_SEMICOLON, ss.str(), peek().line);
97 return std::make_unique<AST::VarAsgnStmt>(var_token.value, std::move(expr), var_token.line);

100AST::StmtPtr Parser::parse_func_decl_stmt() {
101 Token first_token = peek(-1);
102 std::stringstream ss;
103 ss << "Expected function name.\nToken \033[0m'" << peek().value << "'\033[31m is keyword or operator. Please replase it with unique identifier";
104 std::string name = consume(TOK_ID, ss.str(), peek().line).value;
105 std::vector<AST::Argument> args;
106 if (match(TOK_OP_LPAREN)) {
107 while (!match(TOK_OP_RPAREN)) {
108 args.push_back(parse_argument());
109 if (peek().type != TOK_OP_RPAREN) {
110 ss.str("");
111 ss << "Expected \033[0m','\033[31m between function arguments.\nPlease replace \033[0m'";
112 ss << args[args.size() - 1].name << ": " << args[args.size() - 1].type.to_str() << " " << peek().value << "'\033[31m with: \033[0m'"
113 << args[args.size() - 1].name << ": " << args[args.size() - 1].type.to_str() << ", " << peek().value << "'";
114 consume(TOK_OP_COMMA, ss.str(), peek().line);
115 }
116 }
117 }
118
119 AST::Type ret_type = AST::Type(AST::TYPE_NOTH, "noth");
120 if (match(TOK_OP_NEXT)) {
121 ret_type = consume_type();
122 }
123
124 std::vector<AST::StmtPtr> block;
125 consume(TOK_OP_LBRACE, "Expected \033[0m'{'\033[31m after funtion arguments. Prototypes of functions is unsupported in current Topaz compiler version", peek().line);
126 while (!match(TOK_OP_RBRACE)) {
127 block.push_back(parse_stmt());
128 }
129
130 return std::make_unique<AST::FuncDeclStmt>(name, std::move(args), ret_type, std::move(block), first_token.line);
131}

272AST::ExprPtr Parser::parse_primary_expr() {
273 Token token = peek();
274 switch (token.type) {
275 case TOK_OP_LPAREN: {
276 AST::ExprPtr expr = parse_expr();
277 consume(TOK_OP_RPAREN, "Expected ')'. You forgot to specify the closing ')'", token.line);
278 return expr;
279 }
280 case TOK_ID:
281 pos++;
282 if (peek().type == TOK_OP_INC || peek().type == TOK_OP_DEC) {
283 return create_inc_dec_operator(token.value);
284 }
285 return std::make_unique<AST::VarExpr>(token.value, token.line);
286 case TOK_CHARACTER_LIT:
287 pos++;
288 return std::make_unique<AST::CharacterLiteral>(token.value[0], token.line);
289 case TOK_SHORT_LIT:
290 pos++;
291 return std::make_unique<AST::ShortLiteral>(std::stoll(token.value), token.line);
292 case TOK_INT_LIT:
293 pos++;
294 return std::make_unique<AST::IntLiteral>(std::stoll(token.value), token.line);
295 case TOK_LONG_LIT:
296 pos++;
297 return std::make_unique<AST::LongLiteral>(std::stoll(token.value), token.line);
298 case TOK_FLOAT_LIT:
299 pos++;
300 return std::make_unique<AST::FloatLiteral>(std::stold(token.value), token.line);
301 case TOK_DOUBLE_LIT:
302 pos++;
303 return std::make_unique<AST::DoubleLiteral>(std::stold(token.value), token.line);
304 case TOK_BOOLEAN_LIT:
305 pos++;
306 return std::make_unique<AST::BoolLiteral>(token.value == "true", token.line);
307 case TOK_STRING_LIT:
308 pos++;
309 return std::make_unique<AST::StringLiteral>(token.value, token.line);
310 default:
311 throw_exception(SUB_PARSER, "Unsupported expression. Please check expression to mistakes", token.line, token.file_name);
312 }

340AST::Type Parser::consume_type() {
341 Token token = peek();
342 bool is_const = false;
343 bool is_nullable = false;
344 if (match(TOK_CONST)) {
345 is_const = true;
346 }
347 switch (peek().type) {
348 case TOK_BOOL:
349 case TOK_CHAR:
350 case TOK_SHORT:
351 case TOK_INT:
352 case TOK_LONG:
353 case TOK_FLOAT:
354 case TOK_DOUBLE:
355 case TOK_NOTH: {
356 Token type = peek();
357 pos++;
358 if (match(TOK_OP_QUESTION)) {
359 is_nullable = true;
360 }
361 return AST::Type(ttype_to_tvalue(type.type), type.value, is_const, false, is_nullable);
362 }
363 default: {
364 std::stringstream ss;
365 ss << "Token \033[0m'" << peek().value << "'\033[31m is not type. Please replase it to exists type";
366 throw_exception(SUB_PARSER, ss.str(), peek().line, peek().file_name);
